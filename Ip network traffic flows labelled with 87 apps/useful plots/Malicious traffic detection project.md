def ipadress_hosts(ip):
    octets = ip.split('.')
    default_mask = [255,255,255,0]
    i=0
    temp = []
    net_ip = ''
    for octet in octets:
        temp.append(str(default_mask[i] & int(octet)))
        i=i+1
    net_ip = '.'.join(temp)
    return net_ip
